10. Appendix (Code)

library(quanteda)
library(quanteda.textmodels)

data("data_corpus_inaugural")

corpus <- corpus_subset(data_corpus_inaugural, Year >= 2000)

tokens <- tokens(corpus) %>%
  tokens_tolower() %>%
  tokens_remove(stopwords("en"))

dfm_mat <- dfm(tokens)

model <- textmodel_wordfish(dfm_mat)

summary(model)
