## Multiple Linear Regression

lm.fit=lm(medv~.,data=Boston)

library(car)

## Non-linear Transformations

lm.fit2=lm(medv~lstat+I(lstat^2))
summary(lm.fit2)

anova(lm.fit,lm.fit2)

lm.fit5=lm(medv~poly(lstat,5))
summary(lm.fit5)
